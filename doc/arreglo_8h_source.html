190 template<typename Tipo> Moda<Tipo> Arreglo<Tipo>::moda() const {
191  // Obtiene el valor más frecuente
192  using namespace std;
193  vector<Tipo> valores, clases;
194  copia(valores); // Copia valores
195  // Llena clases
196  int i;
197  for (i = 0; i < _tam; ++i) {
198  const int cnt = count(clases.begin(), clases.end(), _valores[i]);
199  if (cnt == 0) {
200  clases.push_back(_valores[i]);
201  }
202  }
203  // Busca la mayor ocurrencia
204  const int tamcss = clases.size();
205  Mayor<Tipo> ma;
206  ma.valor = clases[0];
207  ma.ocurrencias = count(valores.begin(), valores.end(), clases[0]);
208  for (i = 1; i < tamcss; ++i) {
209  const int cnt = count(valores.begin(), valores.end(), clases[i]);
210  if (ma.ocurrencias < cnt) {
211  ma.ocurrencias = cnt;
212  ma.valor = clases[i];
213  }
214  }
215  return Moda<Tipo>(ma);
216 }
